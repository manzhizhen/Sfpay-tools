CREATE TABLE DM_OMS.SMS_RECORD 
(
   ID                   NUMBER(16)           NOT NULL,
   MOBILE_NO            VARCHAR2(20)         NOT NULL,
   MESSAGE              VARCHAR2(2000)       NOT NULL,
   STATUS               VARCHAR2(20),
   MSG_TYPE             VARCHAR2(20),
   SOURCE               VARCHAR2(30),
   BATCH_NO             VARCHAR2(50),
   REMARK               VARCHAR2(200),
   RETURN_CODE          VARCHAR2(50),
   UPDATE_NAME          VARCHAR2(50),
   UPDATE_TIME          DATE,
   CREATE_NAME          VARCHAR2(50),
   CREATE_TIME          DATE
)
PCTFREE 10
INITRANS 1
TABLESPACE OMS_DAT_TS1
LOGGING
MONITORING
 NOPARALLEL;

COMMENT ON TABLE DM_OMS.SMS_RECORD IS
'短信发送记录表';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.ID IS
'类型编码';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.MOBILE_NO IS
'类型名称';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.MESSAGE IS
'短信内容';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.STATUS IS
'短信状态';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.MSG_TYPE IS
'短信类型';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.SOURCE IS
'短信来源';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.BATCH_NO IS
'短信批次号';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.REMARK IS
'备注';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.RETURN_CODE IS
'返回码';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.UPDATE_NAME IS
'修改人';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.UPDATE_TIME IS
'最近更新时间';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.CREATE_NAME IS
'创建人';

COMMENT ON COLUMN DM_OMS.SMS_RECORD.CREATE_TIME IS
'创建时间';
